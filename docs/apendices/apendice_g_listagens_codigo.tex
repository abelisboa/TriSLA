class SemanticParser:
    def __init__(self, ontology):
        self.ontology = ontology

    def parse_intent(self, user_input):
        """
        Converte uma intencao em linguagem natural
        em um conjunto de parametros tecnicos NEST.
        """
        intents = self.ontology.match(user_input)
        return intents

\section{Módulo ML-NSMF}
\begin{lstlisting}[language=Python, caption={Modelo LSTM para previsão de viabilidade de SLA.}, label={lst:ml_nsmf_forecast}]
import numpy as np
from tensorflow.keras.models import Sequential
from tensorflow.keras.layers import LSTM, Dense

def predict_sla_viability(metrics):
    """
    Prediz se o SLA pode ser aceito com base nas metricas de utilizacao.
    """
    model = Sequential([
        LSTM(32, input_shape=(metrics.shape[1], metrics.shape[2])),
        Dense(1, activation='sigmoid')
    ])
    model.compile(loss='binary_crossentropy', optimizer='adam')
    prediction = model.predict(metrics)
    return prediction
\end{lstlisting}

\section{Decision Engine (Motor de Decisão)}
\begin{lstlisting}[language=Python, caption={Fluxo decisório do módulo Decision Engine.}, label={lst:decision_engine}]
class DecisionEngine:
    def __init__(self, model, threshold=0.7):
        self.model = model
        self.threshold = threshold

    def evaluate_sla(self, metrics):
        """
        Avalia a viabilidade do SLA com base na previsao
        de risco e nas metricas observadas em tempo real.
        """
        risk = self.model.predict(metrics)
        if risk < self.threshold:
            decision = "ACCEPT"
        elif self.threshold <= risk < 0.9:
            decision = "RENEGOTIATE"
        else:
            decision = "REJECT"
        return {"risk": float(risk), "decision": decision}
\end{lstlisting}

contract SLAContract {
    address public provider;
    address public tenant;

    bool public violationDetected;

    constructor(uint _latencyThreshold) {
        provider = msg.sender;
        latencyThreshold = _latencyThreshold;
        violationDetected = false;

        if (latency > latencyThreshold) {

\section{Adaptador TriSLA–NASP (I-07)}
\begin{lstlisting}[language=Python, caption={Adaptador TriSLA–NASP para registro de telemetria via REST (I-07).}, label={lst:trisla_nasp_adapter}]
import requests, json
from datetime import datetime

def send_to_nasp(trace_id, module, metrics):
    payload = {
        "traceId": trace_id,
        "module": module,
        "metric": metrics,
        "timestamp": datetime.utcnow().isoformat() + "Z"
    }
    response = requests.post(
        "https://nasp.local/v1/nasp/telemetry",
        headers={"Content-Type": "application/json"},
        data=json.dumps(payload),
        timeout=3
    )
    return response.status_code
\end{lstlisting}

app = Flask(__name__)

@app.route('/v1/agent/ran/reconfigure', methods=['POST'])

    """
    Recebe comandos do Decision Engine (I-06 REST)
    e aplica ajustes em parametros do dominio RAN,
    incluindo potencia de transmissao, QoS e controle de PRBs.

            "qosProfile": qos_profile,
            "prbAllocation": prb_allocation
