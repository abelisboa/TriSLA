\chapter{Diretrizes de Observabilidade e SLOs}
\label{ap:observabilidade}

A arquitetura \ac{trisla} implementa um plano de observabilidade unificado baseado no padrão \textit{OpenTelemetry} (OTLP), abrangendo \textit{traces}, \textit{logs} e métricas em todos os domínios de rede.  
A observabilidade é fundamental para a garantia de \acp{sla}, permitindo o rastreamento ponta a ponta de fluxos e a detecção de anomalias.

\section{Configuração do Pipeline}
\begin{lstlisting}[language=YAML, caption={Configuração mínima do OpenTelemetry Collector.}]
receivers:
  otlp:
    protocols:
      grpc:
      http:
exporters:
  prometheus:
    endpoint: "0.0.0.0:9464"
  logging: {}
service:
  pipelines:
    traces:
      receivers: [otlp]
      exporters: [logging]
    metrics:
      receivers: [otlp]
      exporters: [prometheus, logging]
\end{lstlisting}

\section{Consultas e Indicadores}
\begin{lstlisting}[language=bash, caption={Consultas PromQL para latência e taxa de erro.}]
# Latência p99 por interface
histogram_quantile(0.99, sum(rate(trisla_request_duration_bucket[5m])) by (le, interface))

# Taxa de erro por interface
sum(rate(trisla_requests_total{status=~"5.."}[5m])) by (interface) /
sum(rate(trisla_requests_total[5m])) by (interface)
\end{lstlisting}
