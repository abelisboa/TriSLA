# Comandos para executar no node1 para corrigir o deployment

# 1. Verificar status atual
helm list -n trisla
kubectl get pods -n trisla

# 2. Remover release problemática
helm uninstall trisla-portal -n trisla --wait

# 3. Limpar recursos órfãos
kubectl delete all -n trisla --all --ignore-not-found

# 4. Navegar para o diretório correto
cd ~/gtp5g/trisla-portal

# 5. Reinstalar com configuração corrigida
helm install trisla-portal ./helm/trisla-portal/ \
  -n trisla \
  --create-namespace \
  --wait \
  --timeout=10m

# 6. Verificar se funcionou
kubectl get pods -n trisla
kubectl get services -n trisla
kubectl get deployments -n trisla

# 7. Verificar logs se houver problemas
kubectl logs -n trisla -l app=trisla-portal
